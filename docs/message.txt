  Wallet Generation                                                                                                                                                                                                                                                       
                                                                                                                                                                                                                                                                            
  The wallet keygen API is at src/lib/wallet-keygen.ts. Dependencies needed:                                                                                                                                                                                                
  
  npm install @cosmjs/crypto @cosmjs/encoding bech32                                                                                                                                                                                                                        
                                                                                                                                                                                                                                                                          
  Usage:                                                                                                                                                                                                                                                                    
  
  import { generateWallet, restoreWallet } from "@/lib/wallet-keygen";                                                                                                                                                                                                      
                                                                                                                                                                                                                                                                          
  // Generate new wallet                                                                                                                                                                                                                                                    
  const wallet = await generateWallet();
  wallet.address;   // "bwick1..."                                                                                                                                                                                                                                          
  wallet.mnemonic;  // 24 words                                                                                                                                                                                                                                           
  wallet.pubkey;    // Uint8Array (33 bytes, compressed secp256k1)                                                                                                                                                                                                          
  wallet.privkey;   // Uint8Array (32 bytes)                                                                                                                                                                                                                                
                                                                                                                                                                                                                                                                            
  // Restore from mnemonic                                                                                                                                                                                                                                                  
  const wallet = await restoreWallet("bike elite net siren ...");

  Wallet generation API — already in the codebase at src/lib/wallet-keygen.ts:
  npm install @cosmjs/crypto @cosmjs/encoding bech32
  import { generateWallet, restoreWallet } from "@/lib/wallet-keygen";

  const wallet = await generateWallet();
  // wallet.address  → "bwick1..."
  // wallet.mnemonic → "bike elite net ..."
  // wallet.privkey  → Uint8Array (32 bytes) ← this is what he needs for signing
  // wallet.pubkey   → Uint8Array (33 bytes, compressed secp256k1)

  // To sign transactions with @cosmjs:
  import { DirectSecp256k1Wallet } from "@cosmjs/proto-signing";
  import { SigningStargateClient } from "@cosmjs/stargate";

  const signer = await DirectSecp256k1Wallet.fromKey(wallet.privkey, "bwick");
  const client = await SigningStargateClient.connectWithSigner("http://174.138.87.223:26657", signer);
  // client.sendTokens(), client.signAndBroadcast(), etc.
